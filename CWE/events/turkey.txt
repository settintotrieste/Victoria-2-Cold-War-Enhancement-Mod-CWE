#1950 Turkish Election
country_event = {
	id = 1412189
	title = "$YEAR$ Turkish general election"
	desc = EVTDESC1412189
	picture = "turkey_election_1950"

	fire_only_once = yes

	trigger = { 
		tag = TUR
		year = 1950
		government = presidential_dictatorship
		has_global_flag = blocsenabled
	}

	mean_time_to_happen = { 
		months = 7
	}

	option = {
		name = "Democratic Party"
		
		government = democracy
		country_event = 800054
		ruling_party_ideology = conservative

		any_pop = {
			limit = { AND = { OR = { strata = rich strata = middle } is_accepted_culture = yes } }
			ideology = {
				factor = 0.5
				value = conservative
			}
		}
		
		ai_chance = { factor = 0.8 }
	}

	option = {
		name = "Republican People's Party"

				factor = 0.35
				value = socialist
			}
		}
		
		ai_chance = { factor = 0.1 }
	}

#15th of July Coup
country_event = {
	id = 8015520
	title = EVT_8015520_NAME
	desc = EVTDESC8015520
	picture = "turkey_coup_2016"
	
	fire_only_once = yes
	
	trigger = {
		tag = TUR
		year = 2000
		is_greater_power = no
		has_global_flag = arab_spring
		government = democracy
		average_militancy = 4
		OR = { ruling_party_ideology = conservative ruling_party_ideology = traditionalist ruling_party_ideology = populist }
	}

	mean_time_to_happen = { months = 3 }

	option = {
		name = "Remove these plotters from our ranks!"
		any_pop = { limit = { type = officers } reduce_pop = 0.25 }

		government = presidential_dictatorship
		country_event = 800050
		add_country_modifier = { name = stock_market_crash duration = 365 }
		any_pop = { militancy = -2 }
		any_pop = {
			limit = {
				OR = {
					type = clergymen
					type = bureaucrats
					type = soldiers
				}
				OR = {
					pop_majority_ideology = liberal
					pop_majority_ideology = progressive
				}
			}
			reduce_pop =  0.98
			militancy = 5
			consciousness = 3
		}
		any_pop = {
			scaled_militancy = {
				factor = 10
				ideology = progressive
			}
			scaled_consciousness = {
				factor = 8
				ideology = progressive
			}
			scaled_militancy = {
				factor = 8
				ideology = liberal
			}
			scaled_consciousness = {
				factor = 5
				ideology = liberal
			}
		}
		ai_chance = { factor = 0.7 }
	}
	option = {
		name = EVT_8015521_A
		add_country_modifier = { name = stock_market_crash duration = 900 }
		ruling_party_ideology = liberal

		any_pop = {
			limit = {
				NOT = { religion = secularism }
			}
			scaled_militancy = {
				factor = 10
				ideology = traditionalist
			}
			scaled_consciousness = {
				factor = 8
				ideology = traditionalist
			}
			scaled_militancy = {
				factor = 10
				ideology = populist
			}

			}
			scaled_militancy = {
				factor = 8

			}
		}

	}

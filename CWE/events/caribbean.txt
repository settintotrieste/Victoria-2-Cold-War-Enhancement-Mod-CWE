#Form the West Indies Federation
country_event = {
	id = 1002700
	title = EVT_1002700_NAME
	desc = EVT_1002700_DESC
	picture = "west_indies_federation"
	fire_only_once = yes
	
	trigger = {
		tag = ENG
		owns = 2237 # Trinidad
		OR = { exists = TAN exists = GHA year = 1958 }
		NOT = { year = 1990 }
		war = no
	}

	mean_time_to_happen = { months = 2 }

	option = {
		name = EVT_1002700_A # Form the Federation
		ai_chance = { factor = 0.9 }
		prestige = 25
		any_owned = { limit = { OR = { is_core = AAB is_core = BAR is_core = CYI is_core = DMC is_core = JAM is_core = GRD is_core = STL is_core = SVG is_core = SKN is_core = TAT is_core = TCI province_id = 2225 is_core = VIR } } add_core = XCC }
		release_vassal = XCC
		diplomatic_influence = { who = XCC value = 200 }
	}

	option = {
		name = EVT_1002700_B # Stay integrated part of the empire
		ai_chance = { factor = 0.1 }
		any_pop = { limit = { culture = black_anglo } militancy = 2  consciousness = 2 }
		prestige = -50
	}

}

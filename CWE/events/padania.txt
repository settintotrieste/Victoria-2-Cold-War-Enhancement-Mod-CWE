#Rise of Italian Autonomism
country_event = { 
	id = 8007754
	title = "The Rise of Autonomism"
	desc = EVT_8007754_DESC
	picture = "lega1"
	
	fire_only_once = yes
	
	trigger = {
		tag = ITA
		year = 1983 
		NOT = { has_global_flag = 1992_start_date_flag }
		
		OR = { government = democracy government = democracy1 government = hms_government government = hms_government1 }
		war = no
		is_vassal = no
		
		is_possible_vassal = PAD
	}
	
	mean_time_to_happen = { months = 6 }
	
	option = {
		name = "Interesting..."

		any_owned = {
			limit = { OR = { is_core = PAD is_core = AOS is_core = FVG is_core = VCE } }
			any_pop = { 
				limit = { is_primary_culture = no } 
				ideology = { factor = 0.10 value = nationalist }
			}
		}

 	}
	
}

#The Case for Secession, 1996

	id = 8007756
	title = "$YEAR$ Padanian UDI"
	desc = EVT_8007756_DESC
	picture = "lega3"
	
	fire_only_once = yes
	
	trigger = {
		tag = ITA
		year = 1996
		NOT = { year = 2012 }
		
		OR = { government = democracy government = democracy1 government = hms_government government = hms_government1 }
		war = no
		is_vassal = no
		
		ruling_party_ideology = populist
		
		is_possible_vassal = PAD
	}

		name = "Italy is indivisible!"

			limit = { culture = padanian }

		}

 	}
	
	option = {
		name = "Carve out an autonomous substate!"
		
		#Form the PAD SAR
		release_vassal = PAD

		}

 	}

		release = PAD

			government = democracy

			ruling_party_ideology = conservative
		}

			limit = { culture = italian }

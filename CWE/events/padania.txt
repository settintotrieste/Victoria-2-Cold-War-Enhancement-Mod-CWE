#The Case for Secession, 1996
country_event = { 
	id = 8007756
	title = "$YEAR$ Padanian UDI"
	desc = EVT_8007756_DESC
	picture = "lega3"
	
	fire_only_once = yes
	
	trigger = {
		tag = ITA
		year = 1996
		NOT = { year = 2012 }
		
		OR = { government = democracy government = democracy1 government = hms_government government = hms_government1 }
		war = no
		is_vassal = no
		
		ruling_party_ideology = populist
		
		is_possible_vassal = PAD
	}
	
	mean_time_to_happen = { months = 6 }
	
	option = {
		name = "Italy is indivisible!"
		
		any_pop = { 
			limit = { culture = padanian }
			consciousness = 5
			militancy = 2
			ideology = { factor = 0.05 value = populist }
		}
		
		ai_chance = { factor = 0.9 }
 	}
	
	option = {
		name = "Carve out an autonomous substate!"
		
		#Form the PAD SAR
		release_vassal = PAD
		PAD = {
			government = sar_government
			country_event = 800054
			ruling_party_ideology = conservative
			
			any_pop = { 
				limit = { culture = padanian }
				ideology = { factor = 0.17 value = populist }
			}
		}
		
		sphere_owner = { diplomatic_influence = { who = PAD value = 200 } }
		random_country = {
			limit = { tag = PAD government = sar_government is_vassal = yes }
			relation = { who = THIS value = 400 }
			diplomatic_influence  = { who = THIS value = 200 }
        }
		
		ai_chance = { factor = 0.03 }
 	}
	
	
	option = {
		name = "Proclaim the Republic of the North!"
		
		#Form the PAD
		release = PAD
		PAD = {
			government = democracy
			country_event = 800054
			ruling_party_ideology = conservative
		}
		
		random_country = {
			limit = { tag = PAD government = democracy is_vassal = no }
			relation = { who = THIS value = -400 }
			diplomatic_influence  = { who = THIS value = -200 }
        }
		
		any_pop = { 
			limit = { culture = italian }
			consciousness = 5
			militancy = 5
			ideology = { factor = 0.09 value = nationalist }
			ideology = { factor = 0.03 value = conservative }
		}
		
		ai_chance = { factor = 0.07 }
 	}
	
}

	id = 8017659
	title = "The Fall of Kabul"
	desc = EVTDESC8017659
	picture = "kabul_1992"
	
	fire_only_once = yes

	trigger = {
		tag = AFG
		year = 1992
		NOT = { year = 2002 }
		
		OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 government = socialist_democracy }
		is_vassal = no
		war = no
		
		RUS = { NOT = { OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 government = socialist_democracy } } }
		
		has_country_flag = soviet_withdraw_afg
		has_country_flag = najibullah_afg
		NOT = { has_country_flag = afg_stability }
	}

	mean_time_to_happen = { months = 4 }


	option = {
		name = "The Democratic Republic of Afghanistan Falls"
		clr_country_flag = najibullah_afg
		set_country_flag = islamic_state_afg
		any_pop = { 
			ideology = { factor = 0.10 value = traditionalist }
			militancy = -10
			consciousness = -10
		}
		government = nationalist_dictatorship
		country_event = 800050
		ruling_party_ideology = conservative
		ai_chance = { factor = 0.9 }
	}

	option = {
		name = "The Afghan army and loyal militia rally to defend the capital!"
		add_country_modifier = { name = revolution_chaos duration = 2650 }
		any_pop = { 
			ideology = { factor = 0.15 value = traditionalist }
			militancy = 10
			consciousness = 10
		}
		ai_chance = { factor = 0.1 }
	}
}

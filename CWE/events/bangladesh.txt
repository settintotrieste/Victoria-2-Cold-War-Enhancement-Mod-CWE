#Bangladesh Liberation War
country_event = {
  id = 8017040
  title = "The Bangladesh Liberation War"
  desc = EVTDESC8017040
  picture = "nwo2_bangladesh_war_of_indy"
  
	fire_only_once = yes

	trigger = {
		tag = PAK
		year = 1971
		NOT = { year = 1990 }
		
		is_vassal = no
		war = no
		
		#Awami blocked
		has_country_flag = 1970_PAK_historical_option
		
		#Owns Bangladesh
		is_possible_vassal = BNG
		
		HND = { 
			war = no 
			is_vassal = no 
		}
	}

	mean_time_to_happen = { months = 4 }

	option = {
		name = "Fight to keep East Pakistan!"
		prestige = 50
		
		#Insurgency in Bangladesh
		any_owned = { 
			limit = { is_core = BNG } 
			add_province_modifier = { name = nationalist_agitation duration = 3650 } 
			
			#Boost nationalist support
			any_pop = {
				militancy = 10
				ideology = {
					factor = 0.25
					value = nationalist
				}
			}
			
		}
		
		#Choice for Indian intervention
		HND = { country_event = 8017041 }

		ai_chance = { factor = 0.9 }
	}

	option = {
		name = "Accept Bangladeshi independence"
		prestige_factor = -0.1
		
		#Boost nationalist support
		any_pop = {
			militancy = 7
			ideology = {
				factor = 0.25
				value = nationalist
			}
		}
		
		#Release BNG
		release = BNG
		BNG = { 
			government = democracy 
			country_event = 800054 
			ruling_party_ideology = nationalist 
		}
		sphere_owner = { diplomatic_influence = { who = BNG value = 200 } }
		
		ai_chance = { factor = 0.1 }
	}
}

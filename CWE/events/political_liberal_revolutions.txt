		months = 540
		modifier = {
			factor = 0.1
			has_global_flag = liberal_revolution_somewhere
		}
		modifier = {
			factor = 0.1
			year = 1990
		}
	}

	option = {
		name = "EVTOPTA10001"
		add_country_modifier = {
			name = global_liberal_agitation
			duration = -1
		}

			limit = {

					is_colonial = no

			}

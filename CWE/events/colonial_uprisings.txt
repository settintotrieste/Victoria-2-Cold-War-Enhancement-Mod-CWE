	trigger = {
        is_colonial = yes

			average_militancy = 4 
			AND = { 
				owner = { has_global_flag = blocsenabled }

					pop_majority_ideology = communist 
					pop_majority_ideology = communist_social 
				}
			}
		}

    }

			}
		}

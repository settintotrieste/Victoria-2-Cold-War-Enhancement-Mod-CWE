#1973 Afghan parliamentary election Happen every four years while a monarchy
country_event = {
	id = 80100100
	title = "$YEAR$ Afghan Parliamentary Elections"
	desc = EVTDESC80100100
	picture = "afg_election"
	
	fire_only_once = yes #Effects of this event are too strong so it cannot be a repeating event

	trigger = {
		tag = AFG
		year = 1973
		
		government = hms_government
		is_vassal = no
		
		OR = { ruling_party_ideology = conservative ruling_party_ideology = progressive }
	}

	mean_time_to_happen = { months = 48 }

	option = {
		name = "Conservatives achieve a majority."
		any_pop = { 
			limit = { OR = { strata = middle strata = poor } }
			ideology = { factor = 0.15 value = conservative }
		}
		ai_chance = { 
			factor = 0.5
			modifier = {
				factor = 0.5
				NOT = { pop_majority_ideology = conservative }
			}
		}
	}

	option = {
		name = "The progressive forces prevail."
		any_pop = { 
			limit = { OR = { strata = rich strata = middle } }
			ideology = { factor = 0.15 value = progressive }
		}

			factor = 0.5

				factor = 0.5
